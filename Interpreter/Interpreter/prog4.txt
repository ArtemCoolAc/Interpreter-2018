FUNC LAB
	VARIANT y
	y = PARAM
	y[0,0] = {50}
	VARIANT MULT = {{1}}
	CONVERT DIGIT TO BOOL y[0,0]
	y[1,0] = {TRUE}
	WHILE y[0,0]
		y[0,0]=y[0,0] + {-1}
		y[1,0]=y[1,0] + MULT[0,0]
		MULT[0,0] = MULT[0,0] + {1}
		CONVERT DIGIT TO BOOL y[0,0]
	ENDW
	RETURN y
ENDFUNC

VARIANT x
x=CALL LAB x
CONVERT DIGIT TO STRING x[1,0]
