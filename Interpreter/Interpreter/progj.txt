FUNC LOGIC
	VARIANT y
	y=PARAM
	y = COMMAND "LOOKUP"
	IFWALL y[0,0] RETURN {{0,FALSE}}
	RETURN {{TRUE}}
ENDFUNC

FUNC LAB
	VARIANT z = {{TRUE}}
	VARIANT y
	y = COMMAND "JUMPLEFT PUSHUP PUSHUP RIGHT JUMPRIGHT UP LEFT"
	y = PARAM
	WHILE z[0,0]
		y = COMMAND "UP"
		IFEXIT y[0,0] RETURN y
		z = CALL LOGIC z
		IFZERO z[0,0] y = COMMAND "JUMPUP" IFEXIT y[0,0] RETURN y
		z[0,0] = {TRUE}
	ENDW
	RETURN y
ENDFUNC

VARIANT x
x=CALL LAB x
CONVERT BOOL TO DIGIT x
